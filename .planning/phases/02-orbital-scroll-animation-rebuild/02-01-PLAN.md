    4. **GSAP ScrollTrigger setup (in `useEffect` with cleanup):**
       ```typescript
       useEffect(() => {
         const section = sectionRef.current;
         const pinContainer = pinContainerRef.current;
         if (!section || !pinContainer) return;

         const cards = cardRefs.current.filter(Boolean);
         if (cards.length === 0) return;

         // Get container dimensions for responsive calculations
         const containerWidth = pinContainer.offsetWidth;
         const containerHeight = pinContainer.offsetHeight;
         const centerX = containerWidth / 2;
         const centerY = containerHeight / 2;
         const isMobile = containerWidth < 768;

         // Calculate positions
         const circleRadius = Math.min(Math.min(containerWidth, containerHeight) * 0.35, 350);
         const arcRadius = Math.min(containerWidth, containerHeight * 1.5) * (isMobile ? 1.4 : 1.1);
         const spreadAngle = isMobile ? 100 : 130; // degrees of arc spread

         // Create master timeline
         const tl = gsap.timeline();

         // --- Phase 1: Scatter to Circle (0% - 20% of scroll, per D-06) ---
         cards.forEach((card, i) => {
           const angle = (i / TOTAL_CARDS) * 2 * Math.PI;
           const circleX = centerX + Math.cos(angle) * circleRadius - CARD_WIDTH / 2;
           const circleY = centerY + Math.sin(angle) * circleRadius - CARD_HEIGHT / 2;
           const rotation = (angle * 180 / Math.PI) + 90;

           // Set initial scattered position (random, off-center)
           gsap.set(card, {
             x: centerX + (Math.random() - 0.5) * containerWidth * 1.5 - CARD_WIDTH / 2,
             y: centerY + (Math.random() - 0.5) * containerHeight * 1.0 - CARD_HEIGHT / 2,
             rotation: (Math.random() - 0.5) * 180,
             scale: 0.6,
             opacity: 0,
           });

           // Animate to circle position
           tl.to(card, {
             x: circleX,
             y: circleY,
             rotation: rotation,
             scale: 1,
             opacity: 1,
             ease: "power2.inOut",
             duration: 1,
           }, 0); // all cards morph simultaneously at position 0
         });

         // --- Phase 2: Orbital rotation — 180-degree arc sweep (20% - 100%, per D-06, D-07, D-08) ---
         // Cards sweep through a half-circle arc with cascade/shuffle motion
         const arcApexY = containerHeight * (isMobile ? 0.35 : 0.25);
         const arcCenterY = arcApexY + arcRadius;
         const startAngle = -90 - spreadAngle / 2;
         const endAngle = startAngle + spreadAngle;

         cards.forEach((card, i) => {
           const step = spreadAngle / (TOTAL_CARDS - 1);
           // Starting arc angle for this card
           const cardStartAngle = startAngle + i * step;
           // Ending arc angle — shifted by ~spreadAngle * 0.8 to create cascade
           const cardEndAngle = cardStartAngle - spreadAngle * 0.8;
           const cardStartRad = (cardStartAngle * Math.PI) / 180;
           const cardEndRad = (cardEndAngle * Math.PI) / 180;

           const arcStartX = centerX + Math.cos(cardStartRad) * arcRadius - CARD_WIDTH / 2;
           const arcStartY = arcCenterY + Math.sin(cardStartRad) * arcRadius - CARD_HEIGHT / 2;
           const arcEndX = centerX + Math.cos(cardEndRad) * arcRadius - CARD_WIDTH / 2;
           const arcEndY = arcCenterY + Math.sin(cardEndRad) * arcRadius - CARD_HEIGHT / 2;

           const cardScale = isMobile ? 1.4 : 1.8;

           // Transition from circle to arc start position
           tl.to(card, {
             x: arcStartX,
             y: arcStartY,
             rotation: cardStartAngle + 90,
             scale: cardScale,
             ease: "power1.inOut",
             duration: 0.5,
           }, 1); // starts at timeline position 1 (after morph phase)

           // Sweep through arc (the main orbital motion)
           tl.to(card, {
             x: arcEndX,
             y: arcEndY,
             rotation: cardEndAngle + 90,
             ease: "none", // linear for scrub
             duration: 4, // longest part of timeline
           }, 1.5);
         });

         // --- ScrollTrigger: pin + scrub (per D-01, D-02) ---
         ScrollTrigger.create({
           trigger: section,
           pin: pinContainer,
           start: "top top",
           end: `+=${SCROLL_DISTANCE}`,
           scrub: 1, // smooth 1-second lag for fluid feel
           animation: tl,
           // D-02: auto-unpins when scrub reaches end
         });

         // Cleanup
         return () => {
           ScrollTrigger.getAll().forEach(st => st.kill());
           tl.kill();
         };
       }, [imageUrls]); // re-run if images change
       ```
